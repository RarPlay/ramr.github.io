RAMR in Action

example.rmr
import std.io

// Define a simple data structure
struct Person {
    name: String,
    age: u32,
    email: Option<String>
}

// Pattern matching example
fn greet_person(person: Person) -> String {
    match person {
        Person { name, age: a, email: Some(e) } if a >= 18 => 
            return "Hello " + name + ", your email is " + e,
        Person { name, age: a, email: None } if a >= 18 => 
            return "Welcome " + name + " (adult)",
        Person { name, .. } => 
            return "Hi " + name + " (minor)"
    }
}

// Main function
fn main() {
    let alice = Person {
        name: "Alice",
        age: 25,
        email: Some("alice@example.com")
    };
    
    let bob = Person {
        name: "Bob",
        age: 16,
        email: None
    };
    
    io.println(greet_person(alice));  // "Hello Alice, your email is alice@example.com"
    io.println(greet_person(bob));    // "Hi Bob (minor)"
}
